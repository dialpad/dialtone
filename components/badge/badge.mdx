## Usage

### Import

```jsx
import { DtBadge } from '@dialpad/dialtone-vue';
```

### Default

```jsx
<dt-badge
  text="badge"
/>
```

### Count

```jsx
<dt-badge
  text="1"
  kind="count"
/>
```

### Variants

```jsx
<dt-badge
  text="badge"
  type="default | info | success  | warning | critical | bulletin | ai"
  kind="label | count"
  leftIcon="any icon name"
  rightIcon="any icon name"
/>
```
