## Usage

### Import

```jsx
import { DtDescriptionList } from '@dialpad/dialtone-vue';
```

### Default

```jsx
<dt-description-list
  :items="[{
    term: 'Customer Intent',
    description: 'Hello, I'm looking to return my TV',
  }]"
/>
```

### Column Direction

```jsx
<dt-description-list
  :items="[{
    term: 'Customer Intent',
    description: 'Hello, I'm looking to return my TV',
  }]"
  :direction="column"
/>
```
