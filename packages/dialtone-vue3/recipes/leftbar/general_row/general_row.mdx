## Usage

### Import

```jsx
import { DtRecipeGeneralRow } from '@dialpad/dialtone-vue';
```

### Channel With Unread Messages

```html
<dt-recipe-general-row
  type="channels"
  description="Channel name"
  unread-count="29"
  unread-count-tooltip="29 unread messages"
/>
```

### Contact Center

```html
<dt-recipe-general-row
  type="contact center"
  description="Contact Center name"
  color="magenta-200"
/>
```
