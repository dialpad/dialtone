## Usage

### Import

```jsx
import { DtRecipeCallbarButton } from '@dialpad/dialtone-vue';
```

### Simplest form

```html
<dt-recipe-callbar-button>
  Button
</dt-recipe-callbar-button>
```

### Complex

```html
<dt-recipe-callbar-button
  :disabled="true"
  :active="false"
  :danger="true"
>
  Button
  <template #tooltip>
    This is the tooltip
  </template>
  <template #icon>
    <icon-dialpad-glyph />
  </template>
</dt-recipe-callbar-button>
```
